494  template<typename T>
495  Result ICnc::GetParameterValue(const DataPoint* parameter_datapoint, T& parameter_value, uint32_t policy)
496  {
497  RESULT_START
498  /* Get type name */
499  std::string typeName(typeid(T).name());
500 
501  /* Validate T size corresponds with the DP size */
502  if (sizeof(T) != parameter_datapoint->size)
503  {
504  std::string error_msg{ "Type mismatch asked datapoint type is " + parameter_datapoint->type + " used type is " + typeName };
505  result.error_code = ERROR_CODE_GENERAL;
506  result.error_message = error_msg;
507  throw std::runtime_error(error_msg);
508  }
509  /* Scalar type non-pointer */
510  else
511  {
512  size_t paramTypeLen = sizeof(T);
513  uint8_t* paramByteArray = reinterpret_cast<uint8_t*>(&parameter_value);
514  result = GetParameterById(parameter_datapoint->id, paramTypeLen, paramByteArray, policy);
515  if (result.error_code != ERROR_CODE_OK)
516  return result;
517  }
518  RESULT_END
519  }

523  Result ICnc::GetParameterValue<std::string>(const DataPoint* parameter_datapoint, std::string& parameter_value, uint32_t policy)
524  {
525  RESULT_START

527  uint8_t* strParamArray = new uint8_t[strLen + 1]{ 0 };
528  result = GetParameterById(parameter_datapoint->id, strLen, strParamArray, policy);
529  strParamArray[strLen] = '\0';

531  return result;
